85def build_tree(data: List[bytes], branching_factor: int = 2) -> ctypes.c_void_p:
86 """
87 Build a Merkle tree from a list of byte strings.
88
89 This function wraps the C create_merkle_tree function, handling the complex
90 memory management and type conversions required by ctypes.
91
92 Args:
93 data: List of byte strings to include as leaf nodes in the tree.
94 Each element should be bytes or bytearray.
95 branching_factor: Number of children each internal node can have.
96 Default is 2 (binary tree).
97
98 Returns:
99 ctypes.c_void_p: Opaque pointer to the created merkle_tree_t structure.
100 This pointer must be passed to dealloc_merkle_tree() when done.
101
102 Raises:
103 RuntimeError: If tree creation fails (e.g., out of memory, invalid data).
104 TypeError: If data contains non-bytes elements.
105
106 Example:
107 >>> tree = build_tree([b'hello', b'world', b'merkle', b'tree'])
108 >>> # Use tree for operations...
109 >>> merkle.dealloc_merkle_tree(tree) # Clean up when done
110 """
111 # Validate input
112 if not data:
113 raise ValueError("Data list cannot be empty")
114
115 # Ensure all elements are bytes
116 for i, block in enumerate(data):
117 if not isinstance(block, (bytes, bytearray)):
118 raise TypeError(f"Element {i} must be bytes or bytearray, got {type(block)}")
119
120 count = len(data)
121
122 # Create ctypes array types for the required C arrays
123 # These correspond to the C function signature:
124 # create_merkle_tree(const void **data, const size_t *sizes, size_t count, size_t branching_factor)
125 arr_type = ctypes.c_void_p * count # Array of void pointers for data
126 size_type = ctypes.c_size_t * count # Array of size_t for data lengths
127
128 # Instantiate the arrays
129 data_ptrs = arr_type() # Will hold pointers to each data block
130 sizes = size_type() # Will hold the size of each data block
131
132 # Convert Python bytes objects to C-compatible format
133 buffers = [] # Keep references to prevent garbage collection
134 for i, block in enumerate(data):
135 # Create a C string buffer that contains the data
136 # This allocates memory that C can access
137 buf = ctypes.create_string_buffer(block)
138 buffers.append(buf) # Prevent Python from garbage collecting
139
140 # Cast the buffer to a void pointer and store in our array
141 data_ptrs[i] = ctypes.cast(buf, ctypes.c_void_p)
142 sizes[i] = len(block)
143
144 # Call the C function to create the Merkle tree
145 # The C function will copy the data, so our buffers can be freed after this call
146 tree = merkle.create_merkle_tree(data_ptrs, sizes, count, branching_factor)
147
148 # Check if tree creation succeeded (NULL pointer indicates failure)
149 if not tree:
150 raise RuntimeError('Failed to create Merkle tree - likely out of memory or invalid parameters')
151
152 return tree

155def get_root_hash(tree: ctypes.c_void_p) -> bytes:
156 """
157 Get the root hash of a Merkle tree.
158
159 Args:
160 tree: Opaque pointer to merkle_tree_t returned by build_tree().
161
162 Returns:
163 bytes: 32-byte SHA-256 hash of the tree root.
164
165 Raises:
166 RuntimeError: If hash retrieval fails.
167 ValueError: If tree pointer is invalid.
168
169 Example:
170 >>> tree = build_tree([b'data1', b'data2'])
171 >>> root_hash = get_root_hash(tree)
172 >>> print(f"Root hash: {root_hash.hex()}")
173 """
174 if not tree:
175 raise ValueError("Tree pointer cannot be null")
176
177 # Create a buffer to receive the hash
178 # SHA-256 produces 32-byte hashes
179 hash_buf = (ctypes.c_char * HASH_SIZE)()
180
181 # Call the C function to get the hash
182 result = merkle.get_tree_hash(tree, hash_buf)
183
184 # Check if the operation succeeded
185 if result != MERKLE_SUCCESS:
186 error_messages = {
187 MERKLE_NULL_ARG: "Null argument passed to get_tree_hash",
188 MERKLE_BAD_ARG: "Invalid tree or tree not properly constructed"
189 }
190 error_msg = error_messages.get(result, f"Unknown error code: {result}")
191 raise RuntimeError(f"Failed to get tree hash: {error_msg}")
192
193 # Convert the C char array to Python bytes
194 return bytes(hash_buf.raw)
