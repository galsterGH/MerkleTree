286static merkle_error_t hash_merkle_node(merkle_node_t *parent){
287 // Validate input parameter
288 if (!parent) {
289 return MERKLE_NULL_ARG;
290 }
291
292 // Handle case where node has no children
293 if(parent->child_count == 0 || !parent->children) {
294 return MERKLE_SUCCESS; // No children to hash
295 }
296
297 // Initialize SHA-256 context for combining child hashes
298 SHA256_CTX ctx;
299 SHA256_Init(&ctx);
300
301 // Update hash with each child's hash
302 for(size_t i = 0; i < parent->child_count; ++i){
303 if(!parent->children[i]){
304 return MERKLE_NULL_ARG;
305 }
306
307 SHA256_Update(&ctx, parent->children[i]->hash,HASH_SIZE);
308 }
309
310 // Finalize the hash computation
311 SHA256_Final(parent->hash,&ctx);
312 return MERKLE_SUCCESS;
313}

315static merkle_error_t build_tree_from_queue(queue_t *queue, merkle_tree_t *tree) {
316 // Local variables for managing the next level construction
317 size_t next_level_alloc_count = 0;
318 merkle_node_t **next_level = NULL;
319
320 // Validate input parameters
321 if (!queue || !tree) {
322 return MERKLE_NULL_ARG;
323 }
324
325
326 if(tree->branching_factor == 0){
327 return MERKLE_FAILED_TREE_BUILD;
328 }
329
330 size_t branching_factor = tree->branching_factor;
331
332 /*
333 * Process the queue level by level until a single node remains.
334 * Each iteration collapses one tree level by combining up to
335 * @p branching_factor child nodes under a newly allocated parent.
336 */
337 for (size_t tree_lvl = 0; get_queue_size(queue); tree_lvl++) {
338
339 merkle_node_t *merkle_node = front_queue(queue);
340 size_t queue_size = get_queue_size(queue);
341
342 if(!merkle_node){
343 return MERKLE_FAILED_TREE_BUILD;
344 }
345
346 if (IS_LAST_ELEMENT(queue_size)) {
347
348 if(pop_queue(queue) != merkle_node){
349 clean_up_next_level(&next_level, next_level_alloc_count);
350 return MERKLE_FAILED_TREE_BUILD;
351 }
352
353 tree->root = merkle_node;
354 clean_up_next_level(&next_level, next_level_alloc_count);
355 return MERKLE_SUCCESS;
356 }
357
358 /* Number of parent nodes to create for this level. Each parent will
359 * combine up to @p branching_factor children. */
360 size_t full_nodes = (queue_size + branching_factor - 1)/branching_factor;
361 tree->levels++;
362
363 if(IS_LEAF_NODES_LEVEL(tree_lvl)) {
364 ALLOC_AND_INIT_SIMPLE(next_level, full_nodes);
365
366 if(!next_level){
367 return MERKLE_FAILED_MEM_ALLOC;
368 }
369
370 next_level_alloc_count = full_nodes;
371 }
372
373 // clear all the pointers in next_level to be prepared to process the queue
374 memset(next_level, 0, full_nodes * sizeof(merkle_node_t *));
375
376
377 /* Build the next level of the tree by grouping children under new parents */
378 for(size_t i = 0; i < full_nodes; ++i) {
379 merkle_node_t *parent_node;
380 ALLOC_AND_INIT_SIMPLE(parent_node, 1);
381
382 if(!parent_node){
383 clean_up_next_level(&next_level, next_level_alloc_count);
384 return MERKLE_FAILED_MEM_ALLOC;
385 }
386
387 next_level[i] = parent_node;
388
389 /* Request up to branching_factor children from the queue. The actual
390 * number dequeued is returned in @p dequed. */
391 size_t dequed = branching_factor;
392
393 queue_result_t combo_result = deque_n(
394 queue,
395 &dequed,
396 (void ***)&(parent_node->children));
397
398 if(combo_result != QUEUE_OK){
399 clean_up_next_level(&next_level, next_level_alloc_count);
400 return MERKLE_FAILED_TREE_BUILD;
401 }
402
403 parent_node->child_count = dequed;
404
405 /* Compute the parent's hash from its children. */
406 merkle_error_t ret_code = hash_merkle_node(parent_node);
407
408 for(size_t child_idx = 0; child_idx < dequed; ++child_idx) {
409 parent_node->children[child_idx]->index_in_parent = child_idx;
410 parent_node->children[child_idx]->parent = parent_node;
411 }
412
413 if (ret_code != MERKLE_SUCCESS) {
414 clean_up_next_level(&next_level, next_level_alloc_count);
415 return ret_code;
416 }
417 }
418
419 for(size_t i = 0; i < next_level_alloc_count; ++i){
420
421 /* Enqueue the newly created parent for processing in the next level. */
422 if(next_level[i] && push_queue(queue, next_level[i]) != QUEUE_OK){
423 clean_up_next_level(&next_level, next_level_alloc_count);
424 return MERKLE_FAILED_TREE_BUILD;
425 }
426
427 //next_level[i] should be NULLified since the ownership of the pointer has moved to the queue
428 next_level[i] = NULL;
429 }
430 }
431
432 clean_up_next_level(&next_level, next_level_alloc_count);
433 return MERKLE_SUCCESS;
434}

448merkle_tree_t *create_merkle_tree(const void **data, const size_t *size,
449 size_t count, size_t branching_factor) {
450 merkle_tree_t *tree = NULL;
451 queue_t *queue = NULL;
452
453 // Validate input parameters
454 if (!(data && size && count > 0 && branching_factor > 0)) {
455 return NULL;
456 }
457
458 // Initialize signal protection to catch segfaults gracefully
459 merkle_init_signal_protection();
460
461 // Initialize the tree structure
462 if(init_tree(&tree,count,branching_factor) != MERKLE_SUCCESS){
463 merkle_cleanup_signal_protection();
464 return NULL;
465 }
466
467 TRY{
468
469 // Initialize variables for tree construction
470 size_t leaf_idx = 0;
471 queue = init_queue();
472
473 if(!queue){
474 THROW;
475 }
476
477 bool success = 1;
478 // Process each data block to create leaf nodes
479 for (size_t i = 0; i < count && success; ++i) {
480
481 // Safe access pattern - catch segfaults if count > actual array size
482 SAFE_ACCESS_TRY {
483 // Validate current data block
484 if (!data[i] || !size[i]) {
485 success = 0;
486 break;
487 }
488 } SAFE_ACCESS_CATCH {
489 // Segfault occurred - count parameter is incorrect
490 success = 0;
491 break;
492 } SAFE_ACCESS_END;
493
494 if (!success) break;
495
496 // Allocate memory for new leaf node
497 merkle_node_t *merkle_node;
498 ALLOC_AND_INIT_SIMPLE(merkle_node, 1);
499
500 if(!merkle_node){
501 success = false;
502 break;
503 }
504
505 merkle_node->data = MMalloc(size[i]);
506
507 if(merkle_node->data == NULL){
508 dealloc_hash_node(merkle_node);
509 success = false;
510 break;
511 }
512
513 // Copy the data into the node and compute hash - with protection against invalid data
514 SAFE_ACCESS_TRY {
515 memcpy(merkle_node->data,data[i],size[i]);
516
517 // Compute the hash of the data block
518 if(hash_data_block(data[i], size[i],merkle_node->hash) != MERKLE_SUCCESS){
519 dealloc_hash_node(merkle_node);
520 success = false;
521 break;
522 }
523 } SAFE_ACCESS_CATCH {
524 // Segfault occurred during data access
525 dealloc_hash_node(merkle_node);
526 success = false;
527 break;
528 } SAFE_ACCESS_END;
529
530 if (!success) break;
531
532 // Add the leaf node to the processing queue
533 if(push_queue(queue,merkle_node) != QUEUE_OK){
534 dealloc_hash_node(merkle_node);
535 success = false;
536 break;
537 }
538 // Store reference to the leaf node in the tree
539 (tree->leaves)[leaf_idx++] = merkle_node;
540 }
541
542 if(!success){
543 THROW;
544 }
545
546 RW_WRITE_LOCK(&tree->lock);
547 // Build the internal tree structure from the leaf nodes
548 if (build_tree_from_queue(queue, tree) != MERKLE_SUCCESS) {
549 RW_WRITE_UNLOCK(&tree->lock);
550 THROW;
551 }
552
553 RW_WRITE_UNLOCK(&tree->lock);
554
555 // Clean up the temporary queue and return the completed tree
556 free_queue(queue, dealloc_hash_node);
557 merkle_cleanup_signal_protection();
558 return tree;
559
560 } CATCH();
561
562 free_queue(queue,dealloc_hash_node);
563 clean_up_tree(&tree);
564 merkle_cleanup_signal_protection();
565 return NULL;
566}

620static merkle_error_t generate_proof(merkle_tree_t *const tree, size_t leaf_index, merkle_proof_t **proof){
621
622 // Validate input parameters
623 if(!tree || tree->leaf_count <= leaf_index || !proof){
624 return MERKLE_BAD_ARG;
625 }
626
627 // Initialize local variables
628 merkle_error_t ret = MERKLE_SUCCESS;
629 *proof = NULL;
630 merkle_proof_t *result = NULL;
631
632 TRY{
633
634 ALLOC_AND_INIT_SIMPLE(result,1);
635
636 if(!result) {
637 ret = MERKLE_FAILED_MEM_ALLOC;
638 THROW;
639 }
640
641 result->path_length = tree->levels;
642 result->leaf_index = leaf_index;
643 result->branching_factor = tree->branching_factor;
644
645 ALLOC_AND_INIT_SIMPLE(result->path,tree->levels);
646
647 if(!result->path){
648 ret = MERKLE_FAILED_MEM_ALLOC;
649 THROW;
650 }
651
652 merkle_node_t *node = tree->leaves[leaf_index];
653 bool success = 1;
654
655 for(size_t path_len = 0; path_len < result->path_length && node->parent && success; ++path_len){
656 merkle_node_t *prt = node->parent;
657 ALLOC_AND_INIT(merkle_proof_item_t,proof_item,1);
658
659 result->path[path_len] = proof_item;
660
661 if(!proof_item){
662 success = 0;
663 break;
664 }
665
666 ALLOC_AND_INIT_SIMPLE(proof_item->sibling_hashes, prt->child_count - 1);
667 proof_item->sibling_count = prt->child_count - 1;
668
669 ret = add_proof_path(node->parent,node,proof_item);
670 success = ret == MERKLE_SUCCESS;
671 node = node->parent;
672 }
673
674 if(!success){
675 THROW;
676 }
677
678 *proof = result;
679 return ret;
680
681 }CATCH();
682
683 for(size_t idx = 0; idx != tree->levels; ++idx){
684 if(!result->path[idx]) {
685 continue;
686 }
687
688 MFree(result->path[idx]->sibling_hashes);
689 MFree(result->path[idx]);
690 }
691
692 MFree(result->path);
693 MFree(result);
694 return ret;
695}
